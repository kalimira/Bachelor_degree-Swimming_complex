CREATE DATABASE SWIMMING_COMPLEX;

CREATE TABLE MEASUREMENTS ( time TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, date DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, big_pool FLOAT NOT NULL, kids_pool FLOAT NOT NULL, air_temp FLOAT NOT NULL, uv_index FLOAT NOT NULL );

CREATE TABLE RESERVATIONS ( id SMALLINT NOT NULL, status VARCHAR(10) NOT NULL, name VARCHAR(40), phone_number VARCHAR(20), date VARCHAR(12), booking_time VARCHAR(12), expiring_time VARCHAR(12) );
